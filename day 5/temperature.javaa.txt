public class temperature{
public static void main(String[] args)
{
double[] temps={32.5,28.0,35.6,30.2,29.8};
double highest=temps[0];
double lowest=temps[0];
double sum=0;
for(double t:temps){
sum+=t;
if(t>highest)
{
highest=t;
}
if(t<lowest)
{
lowest=t;
}
}
double average=sum/temps.length;
System.out.println("Highest Temperature= "+highest);
System.out.println("Lowest Temperature= "+lowest);
System.out.println("Average Temperature= "+average);
}
}
